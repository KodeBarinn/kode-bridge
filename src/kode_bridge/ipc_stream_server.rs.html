81/// Message types for streaming
82#[derive(Debug, Clone)]
83pub enum StreamMessage {
84    /// JSON data message
85    Json(Value),
86    /// Text message
87    Text(String),
88    /// Binary data
89    Binary(Bytes),
90    /// Keep-alive ping
91    Ping,
92    /// Server shutdown notification
93    Close,
94}
95
96impl StreamMessage {
97    /// Convert message to bytes for transmission
98    pub fn to_bytes(&self) -> Bytes {
99        match self {
100            Self::Json(value) => {
101                match serde_json::to_vec(value) {
102                    Ok(bytes) => {
103                        let mut output = Vec::with_capacity(bytes.len() + 1);
104                        output.extend_from_slice(&bytes);
105                        output.push(b'\n'); // Line delimiter
106                        Bytes::from(output)
107                    }
108                    Err(_) => Bytes::from("{}\n"),
109                }
110            }
111            Self::Text(text) => {
112                let mut output = text.clone();
113                if !output.ends_with('\n') {
114                    output.push('\n');
115                }
116                Bytes::from(output)
117            }
118            Self::Binary(bytes) => bytes.clone(),
119            Self::Ping => Bytes::from("PING\n"),
120            Self::Close => Bytes::from("CLOSE\n"),
121        }
122    }
123
124    /// Create a JSON message
125    pub fn json<T: serde::Serialize>(value: &T) -> Result<Self> {
126        let json_value = serde_json::to_value(value)
127            .map_err(|e| KodeBridgeError::json_serialize(format!("Failed to serialize: {}", e)))?;
128        Ok(Self::Json(json_value))
129    }
130
131    /// Create a text message
132    pub fn text<T: Into<String>>(text: T) -> Self {
133        Self::Text(text.into())
134    }
135
136    /// Create a binary message
137    pub fn binary<T: Into<Bytes>>(data: T) -> Self {
138        Self::Binary(data.into())
139    }
140}

173/// Statistics for the streaming server
174#[derive(Debug, Clone)]
175pub struct StreamServerStats {
176    /// Total connections accepted
177    pub total_connections: u64,
178    /// Current active connections
179    pub active_connections: u64,
180    /// Total messages sent
181    pub total_messages: u64,
182    /// Total errors
183    pub total_errors: u64,
184    /// Messages per second (approximate)
185    pub messages_per_second: f64,
186    /// Server start time
187    pub started_at: Instant,
188    /// Last stats update time
189    pub last_update: Instant,
190}
191
192impl StreamServerStats {
193    fn new() -> Self {
194        let now = Instant::now();
195        Self {
196            total_connections: 0,
197            active_connections: 0,
198            total_messages: 0,
199            total_errors: 0,
200            messages_per_second: 0.0,
201            started_at: now,
202            last_update: now,
203        }
204    }
205
206    fn update_message_rate(&mut self, message_count: u64) {
207        let now = Instant::now();
208        let duration = now.duration_since(self.last_update).as_secs_f64();
209
210        if duration > 0.0 {
211            self.messages_per_second = message_count as f64 / duration;
212        }
213
214        self.total_messages += message_count;
215        self.last_update = now;
216    }
217}
218
219impl fmt::Display for StreamServerStats {
220    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
221        let uptime = self.started_at.elapsed();
222        write!(
223            f,
224            "Stream Server Stats: {} connections, {} active, {} messages ({:.1}/s), {} errors, uptime: {:?}",
225            self.total_connections,
226            self.active_connections,
227            self.total_messages,
228            self.messages_per_second,
229            self.total_errors,
230            uptime
231        )
232    }
233}

261    pub fn new<F>(generator: F, interval: Duration) -> Self
262    where
263        F: Fn() -> Result<Value> + Send + Sync + 'static,

273impl StreamSource for JsonDataSource {
274    fn next_messages(
275        &mut self,
276    ) -> Pin<Box<dyn Future<Output = Result<Vec<StreamMessage>>> + Send + '_>> {
277        Box::pin(async move {
278            let now = Instant::now();
279            if now.duration_since(self.last_generated) >= self.interval {
280                self.last_generated = now;
281                match (self.generator)() {
282                    Ok(value) => Ok(vec![StreamMessage::Json(value)]),
283                    Err(e) => Err(e),
284                }
285            } else {
286                Ok(vec![])
287            }
288        })
289    }
290
291    fn has_more(&self) -> bool {
292        true // Always has more for periodic generation
293    }
294
295    fn initialize(&mut self) -> Pin<Box<dyn Future<Output = Result<()>> + Send + '_>> {
296        Box::pin(async move { Ok(()) })
297    }
298
299    fn cleanup(&mut self) -> Pin<Box<dyn Future<Output = Result<()>> + Send + '_>> {
300        Box::pin(async move { Ok(()) })
301    }
302}

434    /// Sets the security descriptor for Windows named pipe listener using a SDDL string.
435    ///
436    /// # Arguments
437    /// * `sddl` - Security Descriptor Definition Language string (e.g., `"D:(A;;GA;;;WD)"` for Everyone access).

439    /// # Panics

467    /// Broadcast a message to all connected clients
468    pub fn broadcast(&self, message: StreamMessage) -> Result<usize> {
469        if let Some(ref tx) = self.broadcast_tx {
470            match tx.send(message) {
471                Ok(_) => Ok(tx.receiver_count()),
472                Err(_) => Err(KodeBridgeError::connection("No active receivers")),
473            }
474        } else {
475            Err(KodeBridgeError::connection("Server not started"))
476        }
477    }

479    /// Start the server with a data source
480    pub async fn serve_with_source<S>(&mut self, mut source: S) -> Result<()>
481    where
482        S: StreamSource + 'static,
483    {
484        // Initialize broadcast channel
485        let (broadcast_tx, _) = broadcast::channel(self.config.broadcast_capacity);
486        self.broadcast_tx = Some(broadcast_tx.clone());
487
488        let listener_options = self.listener_options.try_clone()?;
489        let listener = listener_options
490            .name(self.name.clone())
491            .create_tokio()
492            .map_err(|e| KodeBridgeError::connection(format!("Failed to bind server: {}", e)))?;
493
494        info!(" Stream IPC Server listening on {:?}", self.name);
495
496        // Initialize data source
497        source.initialize().await?;
498
499        let (shutdown_tx, mut shutdown_rx) = tokio::sync::oneshot::channel();
500        self.shutdown_tx = Some(shutdown_tx);
501
502        // Start data source task
503        let source_stats = self.stats.clone();
504        let source_broadcast_tx = broadcast_tx.clone();
505        let (source_shutdown_tx, source_shutdown_rx) = tokio::sync::oneshot::channel();
506        let source_shutdown = source_shutdown_rx;
507
508        let source_task = tokio::spawn(async move {
509            let mut source = source;
510            let mut shutdown_rx = source_shutdown;
511
512            loop {
513                tokio::select! {
514                    // Generate data
515                    result = source.next_messages() => {
516                        match result {
517                            Ok(messages) => {
518                                let message_count = messages.len() as u64;
519                                for message in messages {
520                                    if source_broadcast_tx.send(message).is_err() {
521                                        debug!("No receivers for broadcast message");
522                                    }
523                                }
524                                if message_count > 0 {
525                                    source_stats.write().update_message_rate(message_count);
526                                }
527                            }
528                            Err(e) => {
529                                error!("Data source error: {}", e);
530                                source_stats.write().total_errors += 1;
531                            }
532                        }
533
534                        if !source.has_more() {
535                            info!("Data source exhausted");
536                            break;
537                        }
538                    }
539
540                    // Handle shutdown
541                    _ = &mut shutdown_rx => {
542                        info!("Data source shutdown requested");
543                        break;
544                    }
545                }
546
547                // Small delay to prevent tight loops
548                tokio::time::sleep(Duration::from_millis(10)).await;
549            }
550
551            // Cleanup source
552            if let Err(e) = source.cleanup().await {
553                error!("Error cleaning up data source: {}", e);
554            }
555        });
556
557        // Main server loop
558        loop {
559            tokio::select! {
560                // Accept new connections
561                accept_result = listener.accept() => {
562                    match accept_result {
563                        Ok(stream) => {
564                            // Acquire connection permit
565                            if let Ok(permit) = self.connection_semaphore.clone().try_acquire_owned() {
566                                let client_id = self.client_id_counter.fetch_add(1, Ordering::SeqCst);
567
568                                {
569                                    let mut stats = self.stats.write();
570                                    stats.total_connections += 1;
571                                    stats.active_connections += 1;
572                                }
573
574                                let client = StreamClient::new(client_id, format!("client_{}", client_id));
575                                self.clients.write().insert(client_id, client);
576
577                                let config = self.config.clone();
578                                let stats = self.stats.clone();
579                                let clients = self.clients.clone();
580                                let broadcast_rx = broadcast_tx.subscribe();
581
582                                tokio::spawn(async move {
583                                    if let Err(e) = Self::handle_stream_client(
584                                        stream,
585                                        client_id,
586                                        broadcast_rx,
587                                        config,
588                                        stats.clone(),
589                                        clients.clone(),
590                                    ).await {
591                                        error!("Stream client {} error: {}", client_id, e);
592                                        stats.write().total_errors += 1;
593                                    }
594
595                                    // Remove client and update stats
596                                    clients.write().remove(&client_id);
597                                    {
598                                        let mut stats = stats.write();
599                                        stats.active_connections = stats.active_connections.saturating_sub(1);
600                                    }
601
602                                    drop(permit); // Release connection slot
603                                });
604                            } else {
605                                warn!("Maximum connections reached, rejecting new connection");
606                            }
607                        }
608                        Err(e) => {
609                            error!("Failed to accept connection: {}", e);
610                        }
611                    }
612                }
613
614                // Handle shutdown signal
615                _ = &mut shutdown_rx => {
616                    info!("Stream server shutdown requested");
617                    break;
618                }
619            }
620        }
621
622        // Signal source task to stop
623        let _ = source_shutdown_tx.send(());
624
625        // Stop source task
626        source_task.abort();
627
628        // Send close message to all clients
629        let _ = broadcast_tx.send(StreamMessage::Close);
630
631        // Wait for active connections to finish
632        let start = Instant::now();
633        while self.stats.read().active_connections > 0
634            && start.elapsed() < self.config.shutdown_timeout
635        {
636            tokio::time::sleep(Duration::from_millis(100)).await;
637        }
638
639        let remaining = self.stats.read().active_connections;
640        if remaining > 0 {
641            warn!("Shutting down with {} active connections", remaining);
642        }
643
644        info!("Stream IPC Server stopped");
645        Ok(())
646    }

666    /// Handle a single streaming client connection
667    async fn handle_stream_client(
668        mut stream: LocalSocketStream,
669        client_id: u64,
670        mut broadcast_rx: broadcast::Receiver<StreamMessage>,
671        config: StreamServerConfig,
672        stats: Arc<RwLock<StreamServerStats>>,
673        clients: Arc<RwLock<HashMap<u64, StreamClient>>>,
674    ) -> Result<()> {
675        debug!("Handling stream client {}", client_id);
676
677        let mut last_keepalive = Instant::now();
678
679        loop {
680            tokio::select! {
681                // Receive broadcast messages
682                msg_result = broadcast_rx.recv() => {
683                    match msg_result {
684                        Ok(message) => {
685                            match message {
686                                StreamMessage::Close => {
687                                    debug!("Received close message for client {}", client_id);
688                                    break;
689                                }
690                                _ => {
691                                    // Send message to client
692                                    let data = message.to_bytes();
693
694                                    if data.len() > config.max_message_size {
695                                        warn!("Message too large for client {}, skipping", client_id);
696                                        continue;
697                                    }
698
699                                    match timeout(config.write_timeout, stream.write_all(&data)).await {
700                                        Ok(Ok(())) => {
701                                            if let Ok(()) = stream.flush().await {
702                                                // Update client stats
703                                                if let Some(client) = clients.write().get_mut(&client_id) {
704                                                    client.messages_sent += 1;
705                                                    client.last_activity = Instant::now();
706                                                }
707                                            }
708                                        }
709                                        Ok(Err(e)) => {
710                                            error!("Failed to write to client {}: {}", client_id, e);
711                                            if let Some(client) = clients.write().get_mut(&client_id) {
712                                                client.error_count += 1;
713                                            }
714                                            stats.write().total_errors += 1;
715                                            break;
716                                        }
717                                        Err(_) => {
718                                            warn!("Write timeout for client {}", client_id);
719                                            if let Some(client) = clients.write().get_mut(&client_id) {
720                                                client.error_count += 1;
721                                            }
722                                            stats.write().total_errors += 1;
723                                            break;
724                                        }
725                                    }
726                                }
727                            }
728                        }
729                        Err(broadcast::error::RecvError::Lagged(skipped)) => {
730                            warn!("Client {} lagged behind, skipped {} messages", client_id, skipped);
731                            if let Some(client) = clients.write().get_mut(&client_id) {
732                                client.error_count += skipped;
733                            }
734                        }
735                        Err(broadcast::error::RecvError::Closed) => {
736                            debug!("Broadcast channel closed for client {}", client_id);
737                            break;
738                        }
739                    }
740                }
741
742                // Send periodic keepalive
743                _ = tokio::time::sleep(config.keepalive_interval) => {
744                    let now = Instant::now();
745                    if now.duration_since(last_keepalive) >= config.keepalive_interval {
746                        last_keepalive = now;
747
748                        let ping_data = StreamMessage::Ping.to_bytes();
749                        if let Err(e) = timeout(config.write_timeout, stream.write_all(&ping_data)).await {
750                            warn!("Failed to send keepalive to client {}: {:?}", client_id, e);
751                            break;
752                        }
753
754                        if let Err(e) = stream.flush().await {
755                            warn!("Failed to flush keepalive to client {}: {}", client_id, e);
756                            break;
757                        }
758                    }
759                }
760            }
761        }
762
763        debug!("Stream client {} finished", client_id);
764        Ok(())
765    }
766}
