77impl RequestContext {
78    /// Parse request body as JSON
79    pub fn json<T>(&self) -> Result<T>
80    where
81        T: serde::de::DeserializeOwned,
82    {
83        serde_json::from_slice(&self.body).map_err(|e| {
84            KodeBridgeError::json_parse(format!("Failed to parse request JSON: {}", e))
85        })
86    }
87
88    /// Get request body as UTF-8 string
89    pub fn text(&self) -> Result<String> {
90        String::from_utf8(self.body.to_vec()).map_err(|e| {
91            KodeBridgeError::validation(format!("Invalid UTF-8 in request body: {}", e))
92        })
93    }
94
95    /// Get query parameters from URI
96    pub fn query_params(&self) -> HashMap<String, String> {
97        if let Some(query) = self.uri.query() {
98            query
99                .split('&')
100                .filter_map(|pair| {
101                    let mut parts = pair.split('=');
102                    match (parts.next(), parts.next()) {
103                        (Some(key), Some(value)) => Some((
104                            urlencoding::decode(key).ok()?.to_string(),
105                            urlencoding::decode(value).ok()?.to_string(),
106                        )),
107                        (Some(key), None) => {
108                            Some((urlencoding::decode(key).ok()?.to_string(), String::new()))
109                        }
110                        _ => None,
111                    }
112                })
113                .collect()
114        } else {
115            HashMap::new()
116        }
117    }
118
119    /// Get path parameters (when using route patterns)
120    pub fn path_params(&self) -> HashMap<String, String> {
121        // This would be populated by the router during route matching
122        // For now, return empty map
123        HashMap::new()
124    }

169        K: TryInto<http::header::HeaderName>,
170        V: TryInto<http::header::HeaderValue>,
171        K::Error: std::fmt::Debug,

186    /// Set response body from JSON
187    pub fn json<T: serde::Serialize>(self, value: &T) -> Result<Self> {
188        let json_bytes = serde_json::to_vec(value).map_err(|e| {
189            KodeBridgeError::json_serialize(format!("Failed to serialize JSON: {}", e))
190        })?;
191
192        Ok(self
193            .header("content-type", "application/json")
194            .body(json_bytes))
195    }

198    pub fn text<T: Into<String>>(self, text: T) -> Self {
199        self.header("content-type", "text/plain; charset=utf-8")
200            .body(text.into().into_bytes())
201    }

319    /// Add a route with any HTTP method
320    pub fn add_route<F, Fut>(mut self, method: Method, path: &str, handler: F) -> Self
321    where
322        F: Fn(RequestContext) -> Fut + Send + Sync + 'static,
323        Fut: Future<Output = Result<HttpResponse>> + Send + 'static,
324    {

336    /// Find a matching route for the given method and path
337    fn find_route(&self, method: &Method, path: &str) -> Option<&Route> {
338        self.routes
339            .iter()
340            .find(|route| route.method == *method && self.path_matches(&route.path, path))
341    }
342
343    /// Check if a route path matches the request path
344    fn path_matches(&self, route_path: &str, request_path: &str) -> bool {
345        // Validate paths to prevent directory traversal attacks
346        if !self.is_safe_path(request_path) {
347            return false;
348        }
349
350        // Simple exact match for now
351        // TODO: Implement path parameter matching (e.g., /users/{id})
352        route_path == request_path
353    }
354
355    pub fn is_safe_path(&self, path: &str) -> bool {
356        // Reject paths containing directory traversal patterns
357        if path.contains("..") || path.contains("\\") {
358            return false;
359        }
360
361        // Reject paths with null bytes or other control characters
362        if path.contains('\0')
363            || path
364                .chars()
365                .any(|c| c.is_control() && c != '\n' && c != '\r' && c != '\t')
366        {
367            return false;
368        }
369
370        // Ensure path starts with /
371        if !path.starts_with('/') {
372            return false;
373        }
374
375        // Reject excessively long paths
376        if path.len() > 2048 {
377            return false;
378        }
379
380        true
381    }
382}

496    /// Start the server and listen for connections
497    pub async fn serve(&mut self) -> Result<()> {
498        let listener = ListenerOptions::new()
499            .name(self.name.clone())
500            .create_tokio()
501            .map_err(|e| KodeBridgeError::connection(format!("Failed to bind server: {}", e)))?;
502
503        info!("🚀 HTTP IPC Server listening on {:?}", self.name);
504
505        let (shutdown_tx, mut shutdown_rx) = tokio::sync::oneshot::channel();
506        self.shutdown_tx = Some(shutdown_tx);
507
508        loop {
509            tokio::select! {
510                // Accept new connections
511                accept_result = listener.accept() => {
512                    match accept_result {
513                        Ok(stream) => {
514                            // Acquire connection permit
515                            if let Ok(permit) = self.connection_semaphore.clone().try_acquire_owned() {
516                                {
517                                    let mut stats = self.stats.write();
518                                    stats.total_connections += 1;
519                                    stats.active_connections += 1;
520                                }
521
522                                let router = self.router.clone();
523                                let config = self.config.clone();
524                                let stats = self.stats.clone();
525                                let connection_id = {
526                                    let stats = self.stats.read();
527                                    stats.total_connections
528                                };
529
530                                tokio::spawn(async move {
531                                    if let Err(e) = Self::handle_connection(
532                                        stream,
533                                        connection_id,
534                                        router,
535                                        config,
536                                        stats.clone(),
537                                    ).await {
538                                        error!("Connection {} error: {}", connection_id, e);
539                                        let mut stats = stats.write();
540                                        stats.total_errors += 1;
541                                    }
542
543                                    // Connection finished
544                                    {
545                                        let mut stats = stats.write();
546                                        stats.active_connections = stats.active_connections.saturating_sub(1);
547                                    }
548
549                                    drop(permit); // Release connection slot
550                                });
551                            } else {
552                                warn!("Maximum connections reached, rejecting new connection");
553                                // Connection will be dropped, closing it
554                            }
555                        }
556                        Err(e) => {
557                            error!("Failed to accept connection: {}", e);
558                        }
559                    }
560                }
561
562                // Handle shutdown signal
563                _ = &mut shutdown_rx => {
564                    info!("Server shutdown requested");
565                    break;
566                }
567            }
568        }
569
570        // Wait for active connections to finish
571        let start = Instant::now();
572        while self.stats.read().active_connections > 0
573            && start.elapsed() < self.config.shutdown_timeout
574        {
575            tokio::time::sleep(Duration::from_millis(100)).await;
576        }
577
578        let remaining = self.stats.read().active_connections;
579        if remaining > 0 {
580            warn!("Shutting down with {} active connections", remaining);
581        }
582
583        info!("HTTP IPC Server stopped");
584        Ok(())
585    }

594    /// Handle a single client connection
595    async fn handle_connection(
596        mut stream: LocalSocketStream,
597        connection_id: u64,
598        router: Arc<Router>,
599        config: ServerConfig,
600        stats: Arc<RwLock<ServerStats>>,
601    ) -> Result<()> {
602        debug!("Handling connection {}", connection_id);
603
604        let client_info = ClientInfo {
605            connection_id,
606            connected_at: Instant::now(),
607        };
608
609        loop {
610            // Read HTTP request with timeout
611            let request_data =
612                match timeout(config.read_timeout, Self::read_request(&mut stream)).await {
613                    Ok(Ok(Some(data))) => data,
614                    Ok(Ok(None)) => {
615                        debug!("Connection {} closed by client", connection_id);
616                        break;
617                    }
618                    Ok(Err(e)) => {
619                        error!(
620                            "Failed to read request from connection {}: {}",
621                            connection_id, e
622                        );
623                        return Err(e);
624                    }
625                    Err(_) => {
626                        warn!("Read timeout on connection {}", connection_id);
627                        return Err(KodeBridgeError::timeout_msg("Request read timeout"));
628                    }
629                };
630
631            // Parse HTTP request
632            let request_context =
633                match Self::parse_request(request_data, &client_info, config.max_request_size) {
634                    Ok(ctx) => ctx,
635                    Err(e) => {
636                        error!("Failed to parse request: {}", e);
637                        let response = HttpResponse::error(StatusCode::BAD_REQUEST, "Bad Request");
638                        Self::write_response(&mut stream, &response, &config).await?;
639                        stats.write().total_errors += 1;
640                        continue;
641                    }
642                };
643
644            {
645                let mut stats = stats.write();
646                stats.total_requests += 1;
647            }
648
649            if config.enable_logging {
650                info!(
651                    "👤 {} {} {}",
652                    request_context.method, request_context.uri, connection_id
653                );
654            }
655
656            // Clone the method and URI for logging
657            let method = request_context.method.clone();
658            let uri = request_context.uri.clone();
659
660            // Find and execute handler
661            let response = if let Some(route) =
662                router.find_route(&request_context.method, request_context.uri.path())
663            {
664                match timeout(config.write_timeout, (route.handler)(request_context)).await {
665                    Ok(Ok(response)) => response,
666                    Ok(Err(e)) => {
667                        error!("Handler error: {}", e);
668                        HttpResponse::internal_error()
669                    }
670                    Err(_) => {
671                        warn!("Handler timeout");
672                        HttpResponse::error(StatusCode::REQUEST_TIMEOUT, "Handler timeout")
673                    }
674                }
675            } else {
676                HttpResponse::not_found()
677            };
678
679            // Send response
680            match Self::write_response(&mut stream, &response, &config).await {
681                Ok(_) => {
682                    stats.write().total_responses += 1;
683                    if config.enable_logging {
684                        info!("✅ {} {} - {}", method, uri, response.status);
685                    }
686                }
687                Err(e) => {
688                    error!("Failed to write response: {}", e);
689                    stats.write().total_errors += 1;
690                    return Err(e);
691                }
692            }
693
694            // For HTTP/1.0 or if Connection: close, break the loop
695            // For now, we'll keep the connection alive for multiple requests
696        }
697
698        debug!("Connection {} finished", connection_id);
699        Ok(())
700    }

702    /// Read an HTTP request from the stream
703    async fn read_request(stream: &mut LocalSocketStream) -> Result<Option<Vec<u8>>> {
704        let mut buffer = vec![0u8; 8192];
705        let mut total_data = Vec::new();
706
707        loop {
708            match stream.read(&mut buffer).await {
709                Ok(0) => {
710                    // Connection closed
711                    if total_data.is_empty() {
712                        return Ok(None);
713                    }
714                    break;
715                }
716                Ok(n) => {
717                    total_data.extend_from_slice(&buffer[..n]);
718
719                    // Check for end of HTTP headers (double CRLF)
720                    if let Some(header_end) = Self::find_header_end(&total_data) {
721                        // We have complete headers, now check if we need to read body
722                        let headers_str = String::from_utf8_lossy(&total_data[..header_end]);
723
724                        if let Some(content_length) = Self::extract_content_length(&headers_str) {
725                            let body_start = header_end + 4; // Skip the \r\n\r\n
726                            let total_expected = body_start + content_length;
727
728                            // Read remaining body if needed
729                            while total_data.len() < total_expected {
730                                match stream.read(&mut buffer).await {
731                                    Ok(0) => break, // Connection closed unexpectedly
732                                    Ok(n) => total_data.extend_from_slice(&buffer[..n]),
733                                    Err(e) => return Err(KodeBridgeError::Io(e)),
734                                }
735                            }
736                        }
737                        break;
738                    }
739                }
740                Err(e) => return Err(KodeBridgeError::Io(e)),
741            }
742        }
743
744        Ok(Some(total_data))
745    }

747    /// Find the end of HTTP headers (double CRLF)
748    fn find_header_end(data: &[u8]) -> Option<usize> {
749        data.windows(4).position(|window| window == b"\r\n\r\n")
750    }

752    /// Extract Content-Length from headers
753    fn extract_content_length(headers: &str) -> Option<usize> {
754        for line in headers.lines() {
755            if let Some(value) = line
756                .strip_prefix("Content-Length:")
757                .or_else(|| line.strip_prefix("content-length:"))
758            {
759                if let Ok(length) = value.trim().parse::<usize>() {
760                    return Some(length);
761                }
762            }
763        }
764        None
765    }

767    /// Parse raw HTTP request data into RequestContext
768    fn parse_request(
769        data: Vec<u8>,
770        client_info: &ClientInfo,
771        max_size: usize,
772    ) -> Result<RequestContext> {
773        if data.len() > max_size {
774            return Err(KodeBridgeError::validation("Request too large"));
775        }
776
777        // Split headers and body
778        let header_end = Self::find_header_end(&data).ok_or_else(|| {
779            KodeBridgeError::validation("Invalid HTTP request: no header end found")
780        })?;
781
782        let headers_data = &data[..header_end];
783        let body_data = if data.len() > header_end + 4 {
784            &data[header_end + 4..]
785        } else {
786            &[]
787        };
788
789        // Parse request line and headers
790        let headers_str = String::from_utf8_lossy(headers_data);
791        let mut lines = headers_str.lines();
792
793        let request_line = lines
794            .next()
795            .ok_or_else(|| KodeBridgeError::validation("Invalid HTTP request: no request line"))?;
796
797        let mut parts = request_line.split_whitespace();
798        let method = parts
799            .next()
800            .ok_or_else(|| KodeBridgeError::validation("Invalid HTTP request: no method"))?;
801        let uri = parts
802            .next()
803            .ok_or_else(|| KodeBridgeError::validation("Invalid HTTP request: no URI"))?;
804
805        // Parse method
806        let method = Method::from_bytes(method.as_bytes())
807            .map_err(|e| KodeBridgeError::validation(format!("Invalid HTTP method: {}", e)))?;
808
809        // Parse URI
810        let uri = uri
811            .parse::<Uri>()
812            .map_err(|e| KodeBridgeError::validation(format!("Invalid URI: {}", e)))?;
813
814        // Parse headers
815        let mut headers = HeaderMap::new();
816        for line in lines {
817            if let Some((key, value)) = line.split_once(':') {
818                let key = key.trim();
819                let value = value.trim();
820
821                if let (Ok(header_name), Ok(header_value)) = (
822                    key.parse::<http::header::HeaderName>(),
823                    value.parse::<http::header::HeaderValue>(),
824                ) {
825                    headers.insert(header_name, header_value);
826                }
827            }
828        }
829
830        Ok(RequestContext {
831            method,
832            uri,
833            headers,
834            body: Bytes::copy_from_slice(body_data),
835            client_info: ClientInfo {
836                connection_id: client_info.connection_id,
837                connected_at: client_info.connected_at,
838            },
839            timestamp: Instant::now(),
840        })
841    }
842
843    /// Write an HTTP response to the stream
844    async fn write_response(
845        stream: &mut LocalSocketStream,
846        response: &HttpResponse,
847        config: &ServerConfig,
848    ) -> Result<()> {
849        timeout(config.write_timeout, async {
850            // Write status line
851            let status_line = format!(
852                "HTTP/1.1 {} {}\r\n",
853                response.status.as_u16(),
854                response.status.canonical_reason().unwrap_or("Unknown")
855            );
856            stream.write_all(status_line.as_bytes()).await?;
857
858            // Write headers
859            for (key, value) in &response.headers {
860                let header_line = format!("{}: {}\r\n", key, value.to_str().unwrap_or(""));
861                stream.write_all(header_line.as_bytes()).await?;
862            }
863
864            // Write Content-Length if not already present
865            if !response.headers.contains_key("content-length") {
866                let content_length = format!("Content-Length: {}\r\n", response.body.len());
867                stream.write_all(content_length.as_bytes()).await?;
868            }
869
870            // End headers
871            stream.write_all(b"\r\n").await?;
872
873            // Write body
874            if !response.body.is_empty() {
875                stream.write_all(&response.body).await?;
876            }
877
878            stream.flush().await?;
879            Ok::<(), KodeBridgeError>(())
880        })
881        .await
882        .map_err(|_| KodeBridgeError::timeout_msg("Response write timeout"))?
883    }
884}

896// Helper for URL decoding
897pub mod urlencoding {
898    use std::borrow::Cow;
899
900    #[derive(Debug)]
901    pub struct DecodeError;
902
903    impl std::fmt::Display for DecodeError {
904        fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
905            write!(f, "Invalid URL encoding")
906        }
907    }
908
909    impl std::error::Error for DecodeError {}
910
911    pub fn decode(input: &str) -> Result<Cow<'_, str>, DecodeError> {
912        // Safe URL decoding implementation with proper UTF-8 handling
913        if !input.contains('%') && !input.contains('+') {
914            return Ok(Cow::Borrowed(input));
915        }
916
917        let mut result = Vec::new();
918        let bytes = input.as_bytes();
919        let mut i = 0;
920
921        while i < bytes.len() {
922            match bytes[i] {
923                b'%' => {
924                    if i + 2 < bytes.len() {
925                        let hex_str =
926                            std::str::from_utf8(&bytes[i + 1..i + 3]).map_err(|_| DecodeError)?;
927                        let byte = u8::from_str_radix(hex_str, 16).map_err(|_| DecodeError)?;
928                        result.push(byte);
929                        i += 3;
930                    } else {
931                        return Err(DecodeError);
932                    }
933                }
934                b'+' => {
935                    result.push(b' ');
936                    i += 1;
937                }
938                byte => {
939                    result.push(byte);
940                    i += 1;
941                }
942            }
943        }
944
945        // Validate UTF-8 and return
946        let decoded_str = String::from_utf8(result).map_err(|_| DecodeError)?;
947        Ok(Cow::Owned(decoded_str))
948    }
949}
