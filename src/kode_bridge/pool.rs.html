12/// Configuration for connection pool
13#[derive(Debug, Clone, serde::Serialize, serde::Deserialize)]
14pub struct PoolConfig {
15    /// Maximum number of connections in the pool
16    pub max_size: usize,
17    /// Minimum number of idle connections to maintain
18    pub min_idle: usize,
19    /// Maximum time a connection can be idle before being closed (in milliseconds)
20    pub max_idle_time_ms: u64,
21    /// Maximum time to wait for a connection from the pool (in milliseconds)
22    pub connection_timeout_ms: u64,
23    /// Time to wait between connection attempts (in milliseconds)
24    pub retry_delay_ms: u64,
25    /// Maximum number of retry attempts
26    pub max_retries: usize,
27    /// Concurrent request limit
28    pub max_concurrent_requests: usize,
29    /// Rate limiting: max requests per second
30    pub max_requests_per_second: Option<f64>,
31}

65/// A pooled connection wrapper
66pub struct PooledConnection {
67    inner: Option<LocalSocketStream>,
68    created_at: Instant,
69    last_used: Instant,
70    pool: Arc<ConnectionPoolInner>,
71}
72
73impl PooledConnection {
74    fn new(stream: LocalSocketStream, pool: Arc<ConnectionPoolInner>) -> Self {
75        let now = Instant::now();
76        Self {
77            inner: Some(stream),
78            created_at: now,
79            last_used: now,
80            pool,
81        }
82    }
83
84    /// Get the underlying stream
85    pub fn stream(&mut self) -> Option<&mut LocalSocketStream> {
86        self.last_used = Instant::now();
87        self.inner.as_mut()
88    }
89
90    /// Take ownership of the underlying stream
91    pub fn into_stream(mut self) -> Option<LocalSocketStream> {
92        self.inner.take()
93    }
94
95    /// Check if connection is still valid
96    pub fn is_valid(&self) -> bool {
97        self.inner.is_some() && self.last_used.elapsed() < self.pool.config.max_idle_time()
98    }
99
100    /// Get connection age
101    pub fn age(&self) -> Duration {
102        self.created_at.elapsed()
103    }
104
105    /// Get idle time
106    pub fn idle_time(&self) -> Duration {
107        self.last_used.elapsed()
108    }
109}
110
111impl Drop for PooledConnection {
112    fn drop(&mut self) {
113        if let Some(stream) = self.inner.take() {
114            self.pool.return_connection(stream);
115        }
116    }
117}

119/// Internal pool state with PUT request optimization
120struct ConnectionPoolInner {
121    name: Name<'static>,
122    config: PoolConfig,
123    connections: Mutex<VecDeque<(LocalSocketStream, Instant)>>,
124    semaphore: Semaphore,
125    /// 专用于PUT请求的新连接缓存
126    fresh_connections: Mutex<VecDeque<LocalSocketStream>>,
127}
128
129impl ConnectionPoolInner {
130    fn new(name: Name<'static>, config: PoolConfig) -> Self {
131        Self {
132            name,
133            semaphore: Semaphore::new(config.max_size),
134            connections: Mutex::new(VecDeque::new()),
135            fresh_connections: Mutex::new(VecDeque::new()),
136            config,
137        }
138    }
139
140    /// Get a fresh connection for PUT requests, bypassing normal pool
141    async fn get_fresh_connection(&self) -> Result<LocalSocketStream> {
142        // 首先检查是否有预备的新连接
143        {
144            let mut fresh = self.fresh_connections.lock();
145            if let Some(stream) = fresh.pop_front() {
146                return Ok(stream);
147            }
148        }
149
150        // 创建新连接，使用更短的超时和优化的参数
151        let mut last_error = None;
152        for attempt in 0..2 {
153            // 只重试1次，更快失败
154            if attempt > 0 {
155                tokio::time::sleep(Duration::from_millis(10)).await; // 很短的重试延迟
156            }
157
158            match LocalSocketStream::connect(self.name.clone()).await {
159                Ok(stream) => {
160                    debug!("Created fresh connection for PUT request");
161                    return Ok(stream);
162                }
163                Err(e) => {
164                    warn!("Fresh connection attempt {} failed: {}", attempt + 1, e);
165                    last_error = Some(e);
166                }
167            }
168        }
169
170        // 如果新连接失败，回退到池化连接
171        match self.get_pooled_connection() {
172            Some(stream) => {
173                debug!("Falling back to pooled connection for PUT request");
174                Ok(stream)
175            }
176            None => Err(KodeBridgeError::connection(format!(
177                "Failed to get fresh connection and no pooled connections available: {}",
178                last_error
179                    .map(|e| e.to_string())
180                    .unwrap_or_else(|| "Unknown error".to_string())
181            ))),
182        }
183    }
184
185    /// 预热新连接池，为PUT请求做准备
186    async fn preheat_fresh_connections(&self, count: usize) {
187        let mut successful = 0;
188        for _ in 0..count {
189            match LocalSocketStream::connect(self.name.clone()).await {
190                Ok(stream) => {
191                    let mut fresh = self.fresh_connections.lock();
192                    fresh.push_back(stream);
193                    successful += 1;
194                }
195                Err(_) => break,
196            }
197        }
198        if successful > 0 {
199            debug!("Preheated {} fresh connections", successful);
200        }
201    }
202
203    async fn create_connection(&self) -> Result<LocalSocketStream> {
204        let mut last_error = None;
205        let mut delay = self.config.retry_delay();
206
207        for attempt in 0..self.config.max_retries {
208            if attempt > 0 {
209                // Exponential backoff retry delay
210                tokio::time::sleep(delay).await;
211                delay = std::cmp::min(delay * 2, Duration::from_millis(1000));
212            }
213
214            match LocalSocketStream::connect(self.name.clone()).await {
215                Ok(stream) => {
216                    debug!("Created new connection on attempt {}", attempt + 1);
217                    return Ok(stream);
218                }
219                Err(e) => {
220                    warn!("Connection attempt {} failed: {}", attempt + 1, e);
221                    last_error = Some(e);
222                }
223            }
224        }
225
226        Err(KodeBridgeError::connection(format!(
227            "Failed to get fresh connection and no pooled connections available: {}",
228            last_error
229                .map(|e| e.to_string())
230                .unwrap_or_else(|| "Unknown error".to_string())
231        )))
232    }
233
234    fn get_pooled_connection(&self) -> Option<LocalSocketStream> {
235        let mut connections = self.connections.lock();
236
237        // Remove expired connections
238        let now = Instant::now();
239        while let Some((_, created_at)) = connections.front() {
240            if now.duration_since(*created_at) > self.config.max_idle_time() {
241                connections.pop_front();
242            } else {
243                break;
244            }
245        }
246
247        // Get a connection if available
248        connections.pop_front().map(|(stream, _)| {
249            trace!("Reusing pooled connection, {} remaining", connections.len());
250            stream
251        })
252    }
253
254    fn return_connection(&self, stream: LocalSocketStream) {
255        let mut connections = self.connections.lock();
256
257        // Only keep the connection if we haven't exceeded max_size
258        if connections.len() < self.config.max_size {
259            connections.push_back((stream, Instant::now()));
260            trace!("Returned connection to pool, {} total", connections.len());
261        } else {
262            trace!("Pool full, dropping connection");
263        }
264    }
265
266    async fn get_connection_with_timeout(&self) -> Result<LocalSocketStream> {
267        // Try to get a permit within the timeout
268        let permit =
269            tokio::time::timeout(self.config.connection_timeout(), self.semaphore.acquire())
270                .await
271                .map_err(|_| {
272                    KodeBridgeError::timeout(self.config.connection_timeout().as_millis() as u64)
273                })?
274                .map_err(|_| KodeBridgeError::custom("Semaphore closed"))?;
275
276        // Try to get an existing connection first
277        if let Some(stream) = self.get_pooled_connection() {
278            permit.forget(); // Release the permit since we're using a pooled connection
279            return Ok(stream);
280        }
281
282        // Create a new connection
283        let stream = self.create_connection().await?;
284        permit.forget(); // Release the permit
285        Ok(stream)
286    }
287
288    /// Get a fresh connection optimized for PUT requests
289    async fn get_fresh_connection_with_timeout(&self) -> Result<LocalSocketStream> {
290        // Try to get a permit within a shorter timeout for PUT requests
291        let permit = tokio::time::timeout(Duration::from_millis(50), self.semaphore.acquire())
292            .await
293            .map_err(|_| KodeBridgeError::timeout(50))?
294            .map_err(|_| KodeBridgeError::custom("Semaphore closed"))?;
295
296        // Get fresh connection directly
297        let stream = self.get_fresh_connection().await?;
298        permit.forget(); // Release the permit
299        Ok(stream)
300    }
301}

303/// High-performance connection pool for IPC connections
304#[derive(Clone)]
305pub struct ConnectionPool {
306    inner: Arc<ConnectionPoolInner>,
307}
308
309impl ConnectionPool {
310    /// Create a new connection pool
311    pub fn new(name: Name<'static>, config: PoolConfig) -> Self {
312        Self {
313            inner: Arc::new(ConnectionPoolInner::new(name, config)),
314        }
315    }
316
317    /// Create a connection pool with default configuration
318    pub fn with_default_config(name: Name<'static>) -> Self {
319        Self::new(name, PoolConfig::default())
320    }
321
322    /// Get a connection from the pool
323    pub async fn get_connection(&self) -> Result<PooledConnection> {
324        let stream = self.inner.get_connection_with_timeout().await?;
325        Ok(PooledConnection::new(stream, self.inner.clone()))
326    }
327
328    /// Get a fresh connection optimized for PUT requests
329    pub async fn get_fresh_connection(&self) -> Result<PooledConnection> {
330        let stream = self.inner.get_fresh_connection_with_timeout().await?;
331        Ok(PooledConnection::new(stream, self.inner.clone()))
332    }
333
334    /// Preheat fresh connections for better PUT performance
335    pub async fn preheat_for_puts(&self, count: usize) {
336        self.inner.preheat_fresh_connections(count).await;
337    }
338
339    /// Get multiple connections for concurrent operations
340    pub async fn get_connections(&self, count: usize) -> Result<Vec<PooledConnection>> {
341        let mut connections = Vec::with_capacity(count);
342
343        // Use semaphore to control concurrent acquisition
344        let mut tasks = Vec::new();
345        for _ in 0..count {
346            let pool = self.clone();
347            tasks.push(tokio::spawn(async move { pool.get_connection().await }));
348        }
349
350        // Wait for all connection acquisitions to complete
351        for task in tasks {
352            match task.await {
353                Ok(Ok(conn)) => connections.push(conn),
354                Ok(Err(e)) => return Err(e),
355                Err(e) => return Err(KodeBridgeError::custom(format!("Task failed: {}", e))),
356            }
357        }
358
359        Ok(connections)
360    }
361
362    /// Get pool statistics
363    pub fn stats(&self) -> PoolStats {
364        let connections = self.inner.connections.lock();
365        PoolStats {
366            total_connections: connections.len(),
367            available_permits: self.inner.semaphore.available_permits(),
368            max_size: self.inner.config.max_size,
369        }
370    }
371
372    /// Close all pooled connections
373    pub fn close(&self) {
374        let mut connections = self.inner.connections.lock();
375        connections.clear();
376        debug!("Closed all pooled connections");
377    }
378}
